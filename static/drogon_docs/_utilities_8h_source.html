462template <typename T>
463T fromString(const std::string &p) noexcept(false)
464{
465 if constexpr (std::is_integral<T>::value && std::is_signed<T>::value)
466 {
467 std::size_t pos;
468 auto v = std::stoll(p, &pos);
469 // throw if the whole string could not be parsed
470 // ("1a" should not return 1)
471 if (pos != p.size())
472 throw std::invalid_argument("Invalid value");
473 if ((v < static_cast<long long>((std::numeric_limits<T>::min)())) ||
474 (v > static_cast<long long>((std::numeric_limits<T>::max)())))
475 throw std::out_of_range("Value out of range");
476 return static_cast<T>(v);
477 }
478 else if constexpr (std::is_integral<T>::value &&
479 (!std::is_signed<T>::value))
480 {
481 std::size_t pos;
482 auto v = std::stoull(p, &pos);
483 // throw if the whole string could not be parsed
484 // ("1a" should not return 1)
485 if (pos != p.size())
486 throw std::invalid_argument("Invalid value");
487 if (v >
488 static_cast<unsigned long long>((std::numeric_limits<T>::max)()))
489 throw std::out_of_range("Value out of range");
490 return static_cast<T>(v);
491 }
492 else if constexpr (std::is_floating_point<T>::value)
493 {
494 std::size_t pos;
495 auto v = std::stold(p, &pos);
496 // throw if the whole string could not be parsed
497 // ("1a" should not return 1)
498 if (pos != p.size())
499 throw std::invalid_argument("Invalid value");
500 if ((v < static_cast<long double>((std::numeric_limits<T>::min)())) ||
501 (v > static_cast<long double>((std::numeric_limits<T>::max)())))
502 throw std::out_of_range("Value out of range");
503 return static_cast<T>(v);
504 }
505 else if constexpr (internal::CanConvertFromStringStream<T>::value)
506 {
507 T value{};
508 if (!p.empty())
509 {
510 std::stringstream ss(p);
511 // must except in case of invalid value, not return a default value
512 // (else it returns 0 for integers if the string is empty or
513 // non-numeric)
514 ss.exceptions(std::ios_base::failbit);
515 ss >> value;
516 // throw if the whole string could not be parsed
517 // ("1a" should not return 1)
518 if (!ss.eof())
519 std::runtime_error("Bad type conversion");
520 }
521 return value;
522 }
523 else
524 {
525 throw std::runtime_error("Bad type conversion");
526 }
527}

535template <>
536inline bool fromString<bool>(const std::string &p) noexcept(false)
537{
538 if (!p.empty() && std::all_of(p.begin(), p.end(), [](unsigned char c) {
539 return std::isdigit(c);
540 }))
541 return (std::stoll(p) != 0);
542 std::string l{p};
543 std::transform(p.begin(), p.end(), l.begin(), [](unsigned char c) {
544 return (char)tolower(c);
545 });
546 if (l == "true")
547 {
548 return true;
549 }
550 else if (l == "false")
551 {
552 return false;
553 }
554 throw std::runtime_error("Can't convert from string '" + p + "' to bool");
555}

563struct SafeStringHash
564{
565 size_t operator()(const std::string &str) const
566 {
567 const size_t A = 6665339;
568 const size_t B = 2534641;
569 size_t h = fixedRandomNumber;
570 for (char ch : str)
571 h = (h * A) ^ (ch * B);
572 return h;
573 }
